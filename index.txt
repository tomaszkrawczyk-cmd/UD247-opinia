function clean_exit() {
    if (ob_get_length()) ob_end_clean();
    exit;
}

    $mtime = filemtime($faqFile);
    $etagRaw = md5($mtime . filesize($faqFile));
    $etag = '"' . $etagRaw . '"';

    header('Last-Modified: ' . gmdate('D, d M Y H:i:s', $mtime) . ' GMT');
    header('ETag: ' . $etag);

    $hadINM = isset($_SERVER['HTTP_IF_NONE_MATCH']);

    // 1) If-None-Match (Primary Validator)
    if ($hadINM) {
        $inm = trim($_SERVER['HTTP_IF_NONE_MATCH']);

        if ($inm === '*') {
            header($_SERVER['SERVER_PROTOCOL'] . ' 304 Not Modified');
            clean_exit();
        }

        $clientEtags = explode(',', $inm);
        foreach ($clientEtags as $t) {
            $t = trim($t);
            if (stripos($t, 'W/') === 0) {
                $t = trim(substr($t, 2));
            }
            $t = trim(trim($t, '"'));
            if ($t === $etagRaw) {
                header($_SERVER['SERVER_PROTOCOL'] . ' 304 Not Modified');
                clean_exit();
            }
        }
    }

    // 2) If-Modified-Since (tylko jeśli INM nieobecne)
    if (!$hadINM && isset($_SERVER['HTTP_IF_MODIFIED_SINCE'])) {
        $imsHeader = explode(';', $_SERVER['HTTP_IF_MODIFIED_SINCE'])[0];
        $ims = strtotime($imsHeader);
        if ($ims !== false) {
            $now = time();
            if ($ims > $now) $ims = $now;
            if ($mtime <= $ims) {
                header($_SERVER['SERVER_PROTOCOL'] . ' 304 Not Modified');
                clean_exit();
            }
        }
    }
}

/* Explorer launcher: load iframe lazy on first click (CTA button or nav pill) */
(function(){
  function launchExplorer(e){
    if (e) e.preventDefault();
    var wrap = document.getElementById('explorer-iframe-wrap');
    var frame = document.getElementById('foodlaw-explorer-frame');
    var ctaBlock = document.getElementById('explorer-cta-block');
    if (!wrap || !frame) return;
    /* First load: hydrate src from data-src */
    if (!frame.src && frame.dataset.src) {
      frame.src = frame.dataset.src;
    }
    /* Show iframe, hide CTA placeholder */
    wrap.style.display = 'block';
    if (ctaBlock) ctaBlock.style.display = 'none';
    /* Smooth scroll to iframe */
    setTimeout(function(){
      wrap.scrollIntoView({behavior:'smooth', block:'start'});
    }, 50);
  }
  document.addEventListener('DOMContentLoaded', function(){
    /* Bind CTA button */
    var btn = document.getElementById('explorer-launch-btn');
    if (btn) btn.addEventListener('click', launchExplorer);
    /* Bind nav pill (any link to #explorer) */
    var links = document.querySelectorAll('a[href$="#explorer"]');
    for (var i = 0; i < links.length; i++) {
      links[i].addEventListener('click', launchExplorer);
    }
    /* Auto-launch on direct link with #explorer */
    if (window.location.hash === '#explorer') {
      launchExplorer();
    }
  });
})();

  if(!window.chatbase || window.chatbase("getState")!=="initialized"){
    window.chatbase=(...arguments)=>{
      if(!window.chatbase.q){window.chatbase.q=[]}
      window.chatbase.q.push(arguments)
    };
    window.chatbase=new Proxy(window.chatbase,{
      get(target,prop){
        if(prop==="q"){return target.q}
        return(...args)=>target(prop,...args)
      }
    })
  }
  const onLoad=function(){
    const script=document.createElement("script");
    script.src="https://www.chatbase.co/embed.min.js";
    script.id="LsDhyqZTMrkWHaC3dsJRg";
    script.domain="www.chatbase.co";
    document.body.appendChild(script)
  };
  if(document.readyState==="complete"){onLoad()}
  else{window.addEventListener("load",onLoad)}
})();
</script>

<?php if (ob_get_length()) ob_end_flush(); ?>
</body>
</html>
